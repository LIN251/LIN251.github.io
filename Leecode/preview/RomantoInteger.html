13. Roman to Integer

Idea 1:

  1. Dictionary

  2. Backward check

  3. Property:

    right > left
CD = 500-100 = D - C

right = left
II = 1 + 1 = I + I 

right < left
XI = 10 + 1 = X + I

Code:



def romanToint(romanValue):
    romanDic = {"M":1000, "D":500, "C":100, "L":50, "X":10, "V":5, "I":1}

    previous = romanDic.get(romanValue[-1])
    res = previous
    for x in romanValue[-2::-1]:
        current = romanDic.get(x)

        if previous > current:
            res = res - current
        else:
            res = res + current
        previous = current
    return res

if __name__ == '__main__':
    res = romanToint("LVIII")
    print(res)
