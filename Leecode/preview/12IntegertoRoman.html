def intToRoman(intValue):
    romanDic = {1000: "M", 900: "CM", 500: "D", 400: "CD", 100: "C", 90: "XC", 50: "L", 40: "XL", 10: "X", 9: "IX",5: "V", 4: "IV", 1: "I"}
    res = ""
    temp = intValue
    for x in romanDic:
        div = int(temp / x)
        temp = temp % x
        if div == 0:
            continue
        else:
            res = res + romanDic.get(x)*div
        if temp == 0:
            return res
    return res

if __name__ == '__main__':
    res = intToRoman(1994)
    print(res)
