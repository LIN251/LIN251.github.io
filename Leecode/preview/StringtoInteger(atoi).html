def reverse(s):
    result = ""
    s = " ".join(s.split())
    if s == "":
        return 0
    if s[0] == "-" or s[0] == "+":
        if s[0] == "-":
            result += "-"
        s = s[1::1]

    for each in s:
        if each.isdigit():
            result += each
        elif each == "0":
            continue
        elif each == " ":
            break
        else:
            break
    if result == "" or result == "-":
        return 0
    num = int(result)
    if num >= 2147483647:
        return 2147483647
    elif num <= -2147483648:
        return -2147483648
    else:
        return num
